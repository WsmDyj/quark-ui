### 快速上手 Quickstart

```tsx
import React from "react";
import ReactDOM from "react-dom";
import { Countdown } from "@jdcfe/legao";
import "@jdcfe/legao/dist/@jdcfe/legao.css"; //引入样式 如果你选择了按需加载方式则不需要引入
ReactDOM.render(
  <Countdown leftTime={100000}>
    {({ dayStr, hourStr, minuteStr, secondStr }) => (
      <span>
        倒计时剩余： {dayStr}天{hourStr}小时{minuteStr}分钟{secondStr}秒
      </span>
    )}
  </Countdown>,
  document.getElementById("app")
);
```
